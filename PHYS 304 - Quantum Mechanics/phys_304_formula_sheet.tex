\section{Introduction}

\begin{tabular}{@{}ll}
    Schr\"odinger & $i\h\pd\Psi t=-\frac{\h^2}{2m}\ppd\Psi x+V\Psi$ \\
    Probability & $P(a\le x\le b)=\int_a^b\abs{\Psi(x,t)}^2\,dx$ \\
    Expectation & $\agb{x}=\infint x\abs{\Psi(x,t)}^2\,dx$ \\
    Momentum & $\vec p=-i\h\pd{}{x}$ \\
    & $\agb{p}=\infint\Psi^*\lrb{-i\h\pd{}{x}}\Psi\,dx$ \\
    Observable & $\agb{Q(x,p)}=\infint\Psi^*\lrb{Q(x,p)}\Psi\,dx$
\end{tabular}

\section{1-D Schr\"odinger Equation}

\begin{tabular}{@{}ll}
    SOV & $\Psi(x,t)=\psi(x)\vphi(t)$ \\
    & $\vphi(t)=e^{-iEt/\h}$ \\
    TISE & $-\frac{\h^2}{2m}\ddv\psi x+V\psi=E\psi$ \\
    Hamiltonian & $H(x,p)=\frac{p^2}{2m}+V(x)$ \\
    & \twoEqn{H\psi=E\psi}{\agb{H}=E}{7em}{7em} \\
    Wavefunction & $\Psi(x,t)=\sum_n c_n\psi_n(x)e^{-iE_nt/\h}$ \\
    Probability & $P(E_n)=\abs{c_n}^2$
\end{tabular}

\subsection{Harmonic Oscillator $(V=\frac 12kx^2)$}

\subsection{Step Potential $(V=V_0\text{ \normalfont{for} }x>0)$}
